%%% Abstract — Bioinformatics structured form %%%%%%%%%%%%%%%%%%
\begin{abstract}
\noindent\textbf{Motivation:} Ancestry calls for near-isogenic lines (NILs) from
low-cost sequencing normally require calibrating a caller against high-coverage
truth samples, of which few are usually available. We ask whether the calibration
can instead come from cheap simulations.\\
\textbf{Results:} We present nilHMM, a unified duration-aware three-state
(REF/HET/ALT) hidden Markov model with two swappable axes --- the emission (read
counts vs.\ a hard genotype call) and the duration (geometric vs.\ rigidity) ---
that expresses a family of ancestry callers as coordinates in one engine: nNIL,
bbNIL, RTIGER (a Julia-free port), and catiger, alongside a binned Gaussian caller
and an RNA caller (ATLAS). We show that the caller
parameters --- rigidity, error rate, and emission means --- can be calibrated from
Broman \texttt{simcross} simulations rather than from scarce high-coverage truth,
and that the calibrated parameters transfer across skim WGS and 3\textquotesingle{}
RNA-seq (BRB-seq) modalities on real BZea teosinte-introgression NILs. As an
external check, the calibrated callers recover published QTL from an independent
public RIL panel (TeoNAM) after its real genotypes are degraded to low coverage.\\
\textbf{Availability:} The \texttt{nilHMM} R package
(\url{https://sawers-rellan-labs.github.io/nilhmm/}) and the \texttt{zealhmm}
analysis repository (\url{https://sawers-rellan-labs.github.io/zealhmm/}) are
openly available; simulated benchmark data are regenerated with pinned seeds.\\
\textbf{Contact:} \href{mailto:thegemmalab@ncsu.edu}{thegemmalab@ncsu.edu}\\
\textbf{Supplementary information:} Supplementary data are available at
\textit{Bioinformatics} online.
\end{abstract}

\section{Materials and Methods}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\label{sec:materials:methods}

\subsection{A unified duration-aware three-state HMM engine}
One HMM over states REF/HET/ALT (Figure~\ref{fig:engine}A). Two axes are swappable
and define each caller: the \emph{emission} (count/BetaBinomial vs.\ categorical
genotype-confusion) and the \emph{duration/rigidity} layer (geometric, rigidity, or
HSMM). Design start-priors come from the pedigree (BC2S2/BC2S3). The engine is
data-agnostic: $(\text{data}, \text{params}) \rightarrow \text{calls}$. Every named
caller is a coordinate in this (emission $\times$ duration) space
(Figure~\ref{fig:engine}B).

\centering

\caption{\textbf{One engine; every caller has a coordinate in it.}
\textbf{A} The shared duration-aware three-state HMM. Hidden ancestry
$s_t\in\{\mathrm{REF},\mathrm{HET},\mathrm{ALT}\}$ evolves along the marker sequence
with self-transition $1-r$ and transition matrix $A$, emitting the observation $o_t$
(read counts $(n_t,a_t)$ or a hard call $g_t$) through the state-conditional emission
$\psi_s$. The two module boxes are the engine's independent, swappable axes: the
\emph{duration} axis (geometric self-transition, or the rigidity Erlang
state-expansion with minimum run length $\ell$) and the \emph{emission} axis
(BetaBinomial on read counts, or a categorical genotype-confusion), chosen by data
modality. \textbf{B} The named callers as coordinates in the (emission $\times$
duration) grid: \textbf{nNIL} and \textbf{RTIGER} on the published main diagonal, and
bbNIL and catiger on the anti-diagonal (this work's extensions). \textbf{ATLAS} wraps
GOOGA competitive-mapping (allele-specific-expression) thresholding around catiger for
RNA ancestry; binhmm, the pedigree caller, the external ports (LB-Impute, FSFHap), and
the no-HMM control sit off the shared axes.}
\label{fig:engine}
\end{figure*}

\subsection{The callers}
Each named caller is a coordinate in the (emission $\times$ duration) grid
(Figure~\ref{fig:engine}B); the four grid callers differ only along those two axes.
\subsubsection{nNIL} Categorical genotype-confusion emission with a geometric
duration --- Holland's original near-isogenic-line caller, on hard genotype calls
\citex{Holland nNIL}.
\subsubsection{bbNIL} The same geometric duration with a count/BetaBinomial
emission; the low-coverage extension of nNIL to read-count data, where the
self-transition is the smoother.
\subsubsection{RTIGER} Count/BetaBinomial emission with a rigidity (hard
minimum-run) duration in place of the free geometric one; a Julia-free port
\citex{RTIGER}.
\subsubsection{catiger} Categorical emission with the rigidity duration --- the
categorical$+$rigidity caller.
\subsubsection{ATLAS} An RNA caller: because expression and allele-specific
expression confound the competitive-alignment allele fraction (so the BetaBinomial
cannot model it), ATLAS thresholds the competitive counts to hard genotype calls
(GOOGA) and runs catiger --- a wrapper, GOOGA $+$ catiger \citex{GOOGA}.
\subsubsection{binhmm} A three-state Gaussian emission on per-bin allele frequency
with REF anchored at the coverage floor (dense skim only); off the shared axes.

\subsection{Data sources and marker sets}
Skim (low-coverage WGS with teosinte--B73 wideseq filtering), BRB-seq
(3\textquotesingle{} RNA-seq, wideseq-thinned), and MolBreeding target-seq (45K,
high coverage; completeness only). For external validation we additionally use the
public TeoNAM RIL panel \citep{chen2019teonam} --- an independent design (BC1S4
recombinant inbreds, five teosinte parents $\times$ W22), dense genotypes at
51{,}544 AGPv2 SNPs, 22 phenotyped traits, and published QTL.

\subsection{The \texttt{simcross} simulation model}
Forward-simulated BC2S3 (and bulked BC2S2 for skim) NILs on the consensus
genetic map with interference, degraded to each source's depth/error regime
(coverage-driven missingness $\text{missing}(\lambda)=\pi+(1-\pi)e^{-k\lambda}$
and per-read error).

\subsection{Calibration and validation protocol}
Caller parameters are calibrated on \texttt{simcross} truth: segmentation rigidity
by minimum distance to the simulated donor-fragment distribution (not MLE), and
$\text{err}$/emission means on degraded-simulation truth. We then validate on two
tiers. \emph{(i) Simulated truth} --- held-out simulations scored by per-bin state
accuracy, segment-boundary error, breakpoint count, donor-footprint ROC, and
HET-vs-ALT calibration. \emph{(ii) Empirical truth} --- an independent public RIL
panel on which the calibrated pipeline must recover published QTL from
low-coverage-degraded real genotypes (next subsection).

\subsection{External validation on a public RIL panel (TeoNAM)}
Beyond simulated truth, we test whether the calibrated pipeline preserves
downstream mapping signal on \emph{real} genotypes. We treat the TeoNAM hardcall
dosages (0/1/2 $=$ W22\,/\,heterozygous\,/\,teosinte) as truth, lift the AGPv2
marker positions to B73 NAM v5 (v2$\rightarrow$v4$\rightarrow$v5 by chain-file
liftover) and attach cM from the consensus map, then simulate low-coverage read
counts from the true dosages with the same degradation model used for the
\texttt{simcross} benchmark (coverage-driven missingness and per-read error). Each
calibrated caller is run over a coverage sweep, and QTL are re-mapped on the
recovered genotypes with the pipeline applied to the truth, isolating the effect
of degradation and caller error. Recovery is scored for a strongly, consistently
signed domestication trait (staminate spikelet, STAM; five QTL) and a polygenic
agronomic trait (days to anthesis, DTA; nineteen QTL), against both the hardcall
QTL and the published QTL \citep{chen2019teonam}. This is a real-genotype,
simulated-read test --- it probes the genotype$\rightarrow$degrade$\rightarrow$
call$\rightarrow$map chain under realistic recombination and trait architecture,
not raw low-coverage resequencing.

\subsection{QTL recovery on a public RIL panel (TeoNAM)}
Whether the simulation-calibrated callers preserve downstream mapping signal on an
independent real panel, as a function of coverage. QTL mapped from the callers'
low-coverage-degraded genotypes are compared to those from the hardcall truth by
QTL sensitivity (recovery within the support interval), false-positive QTL,
peak-position error, LOD attenuation, and additive-effect sign concordance. STAM
(five strongly, consistently signed QTL) gives a sharp recovery-vs-coverage
threshold; the polygenic DTA (nineteen QTL) is the harder case, and both are
cross-checked against the published joint-linkage QTL \citep{chen2019teonam}.

As a first, sharp instance of this recovery test, the STAM coverage sweep
(Figure~\ref{fig:stam-sweep}) contrasts the two duration-aware HMM callers with a
deliberately heterozygous-prone no-HMM control --- a per-site
genotype-likelihood caller under a Hardy--Weinberg prior, for which a single
alternate read is already called heterozygous. Because ancestry is a long-block
signal, both HMM callers reconstruct the STAM joint-linkage peaks essentially at
their full-coverage height down to $0.1\times$: the \textit{tb1} peak holds near
$-\log_{10}P\approx17$ across the whole sweep (bbNIL never below $17.1$; RTIGER
dips only to $16.1$ at $0.1\times$, from a $\lambda=\infty$ baseline of $17.3$),
and the heterozygous-call fraction stays at the $\sim$8\% truth level. The no-HMM
control, deciding each site independently, instead inflates heterozygous calls to
$\sim$20\% at $\lambda\le1\times$ (het excess) and attenuates the \textit{tb1}
peak to $-\log_{10}P\approx13.4$ at $0.1\times$. The duration/linkage prior, not
the genotype likelihood, is what confers low-coverage robustness.
